– Функція оновлення списку координат
local function updateCoordsList()
for i, child in pairs(CoordsListFrame:GetChildren()) do
if child:IsA(“TextButton”) then
child:Destroy()
end
end

```
for i, coord in pairs(savedCoordinates) do
    local coordButton = Instance.new("TextButton", CoordsListFrame)
    coordButton.Size = UDim2.new(0, 280, 0, 25)
    coordButton.Position = UDim2.new(0, 10, 0, (i-1) * 30)
    coordButton.Text = coord.name .. " (" .. coord.x .. ", " .. coord.y .. ", " .. coord.z .. ")"
    coordButton.BackgroundColor3 = Color3.fromRGB(100, 100, 200)
    coordButton.TextScaled = true
    
    -- Телепорт на координати
    coordButton.MouseButton1Click:Connect(function()
        if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
            LocalPlayer.Character:MoveTo(Vector3.new(coord.x, coord.y, coord.z))
        end
    end)
    
    -- Видалення координат (правий клік)
    coordButton.MouseButton2Click:Connect(function()
        table.remove(savedCoordinates, i)
        updateCoordsList()
    end)
end

CoordsListFrame.CanvasSize = UDim2.new(0, 0, 0, #savedCoordinates * 30)
```

end

– Логіка переміщення меню
MenuFrame.InputBegan:Connect(function(input)
if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
isDraggingMenu = true
dragStart = input.Position
startPos = MenuFrame.Position
end
end)

MenuFrame.InputEnded:Connect(function(input)
if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
isDraggingMenu = false
end
end)

UserInputService.InputChanged:Connect(function(input)
if isDraggingMenu and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
local delta = input.Position - dragStart
MenuFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end
end)
